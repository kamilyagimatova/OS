#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>


void* start_thread() {
    printf("Hello from thread %d!\n", (int)pthread_self());
	pthread_exit(NULL);
}

int main() {
    int n = 10;
    pthread_t threads[n];
    for (int i = 0; i < n; i++) {
        pthread_create(&threads[i], NULL, start_thread, NULL);
        printf("Thread %d created.\n", (int)threads[i]);
    }
    pthread_exit(NULL);
    return 0;
}
